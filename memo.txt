이걸 알아야지 너랑 front나 다른 기능을 개발할때 좋을것 같아.



(function() {
    console.log("✨ [시각적 요소 추적기] 활성화! 클릭 시 1초간 빨간 테두리가 표시됩니다.");  
    document.addEventListener('click', function(e) {
        if (e.target.closest('#eruda') || e.target.tagName === 'HTML') return;        
        const el = e.target;
        
        // 시각적 하이라이트 효과 (1초간 빨간 테두리)
        const oldBorder = el.style.border;
        el.style.border = '2px solid #FF0000';
        setTimeout(() => { el.style.border = oldBorder; }, 1000);
        
        // 정보 출력
        const tag = el.tagName.toLowerCase();
        const id = el.id ? `#${el.id}` : '';
        const cls = el.className && typeof el.className === 'string' ? `.${el.className.trim().split(/\s+/).join('.')}` : '';
        console.log(`${tag}${id}${cls}`);
    }, true);
})();
